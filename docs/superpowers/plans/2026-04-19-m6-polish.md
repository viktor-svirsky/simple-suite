- [ ] **Step 2: Implement**

```swift
import UIKit

enum Haptics {
    enum Kind {
        case pin, delete, success
    }

    @MainActor
    static func fire(_ kind: Kind) {
        switch kind {
        case .pin:
            let g = UIImpactFeedbackGenerator(style: .soft)
            g.prepare(); g.impactOccurred()
        case .delete:
            let g = UINotificationFeedbackGenerator()
            g.prepare(); g.notificationOccurred(.warning)
        case .success:
            let g = UINotificationFeedbackGenerator()
            g.prepare(); g.notificationOccurred(.success)
        }
    }
}
```

Call sites: `NoteActions.togglePin` → `Haptics.fire(.pin)`, `NoteActions.delete` → `Haptics.fire(.delete)`.

## Task 3: Dynamic Type audit

**Files:**
- Modify: `apps/simple-notes/SimpleNotes/Theme/Theme.swift` — make serif/mono/sans scale

- [ ] **Step 1: Update helpers**

```swift
enum Font {
    static let serifName = "New York"
    static let sansName  = "SF Pro Text"
    static let monoName  = "SF Mono"

    static func serif(_ size: CGFloat, weight: SwiftUI.Font.Weight = .regular) -> SwiftUI.Font {
        .custom(serifName, size: size, relativeTo: .body).weight(weight)
    }
    static func sans(_ size: CGFloat, weight: SwiftUI.Font.Weight = .regular) -> SwiftUI.Font {
        .system(.body, design: .default, weight: weight)
            .leading(.tight)
            // size override via relativeTo:
    }
    static func mono(_ size: CGFloat) -> SwiftUI.Font {
        .system(.caption, design: .monospaced)
    }
}
```

(Implementation note: prefer `.custom(name:size:relativeTo:)` for serif; for sans/mono use `.system(textStyle:design:)` so Dynamic Type scales the text style. The size param in sans/mono becomes a soft hint — pick the closest text style that matches each site.)
